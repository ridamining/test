async function fetchSheetData() { const tbody = document.getElementById('adminTableBody'); tbody.innerHTML = 'CONNECTING TO CLOUD...'; // Add a 10-second timeout safety const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), 10000); try { const response = await fetch(WEB_APP_URL, { method: 'GET', signal: controller.signal }); if (!response.ok) throw new Error('Network response was not ok'); const data = await response.json(); clearTimeout(timeoutId); if (!data || data.length < 2) { tbody.innerHTML = 'Registry Empty: No Users Found in Sheet'; return; } // Render Data (skipping the header row) tbody.innerHTML = data.slice(1).map((row) => ` ${row[1] || 'N/A'}
${row[0] || ''} ${row[3] || 'UNVERIFIED'} `).join(''); } catch (err) { console.error("Fetch Error:", err); tbody.innerHTML = `

Connection Failed

${err.message}

`; } }
